struct binary_tree  //二叉树
{
    int data;  //数据域
    struct binary_tree* child_left;  //左子节点
    struct binary_tree* child_right;  //右子节点
};

创建

struct binary_tree* Create_Tree()
{
    struct binary_tree* Tree_top = (struct binary_tree*)malloc(sizeof(struct binary_tree));  //为新结点分配动态内存空间
    int tree_data;  //新数据
    scanf_s("%d", &tree_data);
    if (tree_data != 0)  //这里的限制条件是如果数据是 0，就不存在这个结点
    {
        Tree_top->data = tree_data;
        Tree_top->child_left = Create_Tree();  //以左子节点为左树的树顶
        Tree_top->child_right = Create_Tree();  //以右子节点为右树的树顶
    }
    else
        return NULL;
    return Tree_top;
}　

遍历算法

前序遍历

前序遍历算法是从树顶然后分向左树至右树一一输出

void Binary_tree_Print(struct binary_tree* tree_top)  //前序遍历算法
{
    if (tree_top == NULL)  //当前树为空
        return;
    printf("%d", tree_top->data);  //输出当前树的值
    Binary_tree_Print(tree_top->child_left);  //输出左树
    Binary_tree_Print(tree_top->child_right);  //输出右树
}

中序遍历

中序遍历是从左树返回树顶然后到右树输出

void Binary_tree_Print_In(struct binary_tree* tree_top)  //中序遍历算法
{
    if (tree_top == NULL)  //当前树为空
        return;
    Binary_tree_Print_In(tree_top->child_left);  //输出左树
    printf("%d ", tree_top->data);  //输出当前树的值
    Binary_tree_Print_In(tree_top->child_right);  //输出右树
}

后序遍历

后序遍历算法是从左到右先叶后顶输出

void Binary_tree_Print_Post(struct binary_tree* tree_top)  //后序遍历算法
{
    if (tree_top == NULL)  //当前树为空
        return;
    Binary_tree_Print_Post(tree_top->child_left);  //输出左树
    Binary_tree_Print_Post(tree_top->child_right);  //输出右树
    printf("%d ", tree_top->data);  //输出当前树的值
}

线索二叉树

只是多了两个叶节点判断标志而已

结构定义及建立

enum Point_tag  //叶节点状态,Link值为 0 代表指向左右子节点,Thread值为 1 代表指向前驱或者后继的线索
{
    Link, Thread
};

struct treaded_binary_tree  //线索二叉树
{
    int data;  //数据域
    struct treaded_binary_tree* child_left;  //左子节点
    struct treaded_binary_tree* child_right;  //右子节点
    bool tag_left;  //左标志
    bool tag_right;  //右标志
};

struct treaded_binary_tree* Create_Binary_Tree()  
{
    struct treaded_binary_tree* Tree_top = (struct treaded_binary_tree*)malloc(sizeof(struct treaded_binary_tree));  //动态二维空间
    int tree_data;  //数据
    scanf_s("%d", &tree_data);
    if (tree_data != 0)  //不为 0 时
    {
        Tree_top->data = tree_data;
        Tree_top->child_left = Create_Binary_Tree();
        if (Tree_top->child_left != NULL)  //左标志标记
            Tree_top->tag_left = Link;
        Tree_top->child_right = Create_Binary_Tree();
        if (Tree_top->child_right != NULL)  //右标志标记
            Tree_top->tag_right = Link;
    }
    else
        return NULL;
    return Tree_top;
}

中序遍历线索化

线索化之前需要定义一个指向根结点（即树顶）的头结点Head，就像双向链表的结构一样

struct treaded_binary_tree* t;  //全局变量，指向刚才访问过的结点

void Binary_tree_Threaded_In(struct treaded_binary_tree* Tree_top)  //无头结点的中序线索化
{
    if (Tree_top != NULL)  //不为空
    {
        Binary_tree_Threaded_In(Tree_top->child_left);  //左树
        if (Tree_top->child_left == NULL)  //无左树
        {
            Tree_top->tag_left = Thread;  //前驱线索
            Tree_top->child_left = t;  //左指针指向前驱
        }
        if (t->child_right == NULL)  //前驱无右树
        {
            t->tag_right = Thread;  //后继线索
            t->child_right = Tree_top;  //前驱右指针指向后继
        }
        t = Tree_top;  //保持 t 指向Tree_top的前驱
        Binary_tree_Threaded_In(Tree_top->child_right);  //右树
    }
}

struct treaded_binary_tree* Head_Treading(struct treaded_binary_tree* Binary_tree_treaded)  //中序线索化主体，返还头结点指针
{
    struct treaded_binary_tree* Head = (struct treaded_binary_tree*)malloc(sizeof(struct treaded_binary_tree));  //动态内存
    t = Binary_tree_treaded;  //t一开始指向根结点
    Binary_tree_Threaded_In(Binary_tree_treaded);  //无头结点的中序线索化，在后面加入头结点进行调整
    Head->child_left = Binary_tree_treaded;  //头结点左指针指向根结点
    struct treaded_binary_tree* q = Binary_tree_treaded;  //分配 q 来作为第一个结点（第一个结点就是中序遍历中第一个整的结点）
    while (q->tag_left == Link)
        q = q->child_left;
    q->child_left = Head;  //第一个结点的左指针指向头结点（一开始是指向根结点）
    struct treaded_binary_tree* p = Binary_tree_treaded;  //分配 p 来作为最后一刻个结点（最后一个结点就是中序遍历中最后一个整的结点）
    while (p->tag_right == Link)
        p = p->child_right;
    Head->child_right = p;  //最后一个结点的右指针指向头结点（一开始哪里也没指）
    p->child_right = Head;  //头结点的右指针指向最后一个结点（一开始哪里也没指）
    Head->tag_left = Link;  //头结点的左指针指向的是根结点
    Head->tag_right = Thread;  //头结点的右指针是前驱后继
    return Head;

线索二叉树的遍历输出

void Binary_tree_Threaded_Print(struct treaded_binary_tree* head)
{
    struct treaded_binary_tree* p = head->child_left;  //p指向根结点
    while (p != head)  //空树或者遍历结束时，p == head
    {
        while (p->tag_left == Link)  //循环到中序序列的第一个结点
            p = p->child_left;
        printf("%d ", p->data);  //输出
        while (p->tag_right == Thread && p->child_right != head)  //前驱右指针指向后继时，并且后继不为头结点时
        {
            p = p->child_right;  //前往右指针指向后继
            printf("%d ", p->data);  //输出
        }
        p = p->child_right;  //p前往右树
    }
}

树、二叉树与森林

树转换为二叉树

  1. 加线：在所有兄弟节点之间加一条连线
  2. 去线：对树中每一个结点，只保留它与第一个孩子结点的连线，删除它与其它孩子结点之间的连线
  3. 层次调整：大致意思为将树的样子变得好看一些，层次更加鲜明一些，注意第一个孩子是二叉树结点的左孩子，兄弟转换过来的孩子是结点的右孩子

森林转换为二叉树

  1. 把每棵树转换为二叉树
  2. 第一棵二叉树不动，从第二颗起，依次把后一棵二叉树的根节点作为前一棵二叉树的根节点的右孩子

二叉树转换为树

  1. 加线：若某结点的左孩子结点存在，则将这个左孩子的右孩子结点、左孩子的右孩子的右孩子结点…….

左孩子的n个右孩子结点都作为此结点的孩子，用线连接

  1. 去线：删除原二叉树中所有结点与其右孩子结点的连线
  2. 层次调整：略过……

二叉树转换为森林

判断一棵二叉树能否转换为森林还是树，只需要看这棵二叉树的根结点有没有右孩子，有就是森林，没有就是一棵树

  1. 从根节点开始，若右孩子存在，则把与右孩子结点的连线删除，再看分离后的二叉树，再进行操作，直到所有的与根节点相关的右孩子连线都被删除
  2. 再将每棵分离后的二叉树转换为树即可

树与森林的遍历

树的遍历

先根遍历

先访问根结点，然后依次遍历根的每颗子树

后根遍历

先依次访问每颗子树，然后访问根结点

森林的遍历

前序遍历

先访问第一棵树的根结点，然后依次先根遍历，然后遍历剩下的森林

后序遍历

先访问第一棵树，用后根遍历，然后遍历剩下的森林

我们不难发现，森林的前序遍历和二叉树的前序遍历结果相同，森林的后序遍历和二叉树的中序遍历结果相同

二叉树应用

赫曼夫树

带权路径长度WPL最小的二叉树称之为赫曼夫树

构造算法

  1. 根据给定的n个权值{w1,w2······wn}构成n棵二叉树的集合F={T1,T2······Tn}，其中每棵二叉树Ti中只有一个带权为wi的根结点，其左右子树均为空
  2. 在F中选取两棵根结点权值最小的树作为左右子树构造一棵新的二叉树，且置信的二叉树的根结点的权值为其左右子树上根节点的权值之和
  3. 在F中删除这两棵树，同时将新得到的二叉树加入F中
  4. 重复2、3直到F中只有一棵树为止，这棵树就是赫曼夫树

快速排序

快速排序是啥，是一个很顶的排序，二话不多说，上代码以示友好

#include <stdio.h>

/*基值在一个区块中，左边的数要小，右边的数要大*/

int a[15];  //定义长度

int Quick_patition(int left, int right);  //申明

void swap(int u, int v)  //交换数组中下标为u、v的值
{
    int t = a[u];
    a[u] = a[v];
    a[v] = t;
}

void Quick_Sort(int left, int right)  //对当前区块（从left到right这一段）进行操作
{
    if (right - left <= 0)  //如果该区块的长度已经小于等于 0，结束对这个区块的操作
        return;
    int pivot = Quick_patition(left, right);  //得到这个区块的基值所在的位置
    Quick_Sort(left, pivot - 1);  //对这一个区块中分出来的左区块进行操作
    Quick_Sort(pivot + 1, right);  //对这一个区块分出来的有区块进行操作
}

int Quick_patition(int left, int right)  //left是左标在数组的小标，right是右标在数组的下标（这个函数用以得到基值在当前区块的位置）
{
    int base = right--;  //定义区块的基值为最右边的数，base为基值所在的下标
    while (true)  //只有左标移动到了右标的位置处才会结束循环
    {
        while (a[left] < a[base])  //先移动左标，比较左标与基值，如果左标小于基值，左标右移，直到左标大于等于基值
            left++;
        while (a[right] > a[base])  //再移动右标，比较右标与基值，如果右标大于基值，右标左移，直到右标小于等于基值
            right--;
        if (left >= right)  //如果左标已经移动到了右标处甚至右标的右边，则结束
            break;
        else  //如果左标还没有移动到右标的位置，就交换左标与右标
            swap(left, right);
    }
    swap(left, base);  //交换左标与基值，使基值位于该区块中间
    return left;  //返还左标的下标
}

int main(void)
{
    int n; //数组中值个数
    scanf_s("%d", &n);
    for (int i = 0; i < n; i++)  //赋值
        scanf_s("%d", &a[i]);
    Quick_Sort(0, n - 1);  //快速排序
    for(int i = 0; i < n; i++)  //打印
        printf("%d ", a[i]);  
    printf("\n");
    return 0;
}

队列

队列和栈刚好相反嗷，它是先进先出的线性表（FIFO）

允许删除的一端称为队头（front），允许插入的一端称为对尾（rear）

队列的顺序存储结构

由于数组存储空间的问题（即假溢出），我们使用的队列应该是头尾相接的循环队列，不然很容易就会超出数组范围

顺带一提队列长度计算公式：( rear - front + QueueSize ) % QueueSize

结构定义及初始化

#define maxn 1000
#define ERROR 0
#define RIGHT 1
#define Status int

struct Queue  //循环队列的顺序结构
{
    int data[maxn];  //队列中元素
    int front;
    int rear;
};

Status Queue_Init(struct Queue* queue)  //队列初始化
{
    queue->front = 0;  //队头归零
    queue->rear = 0;  //队尾归零
    return RIGHT;
}

int main(void)
{
    struct Queue SqQueue;
    if (Queue_Init(&SqQueue) != RIGHT)  //队列创建
        return;
}

入队

Status Queue_Enter(struct Queue* queue, int elem)  //入队，elem为入队元素
{
    if ((queue->rear + 1) % maxn == queue->front)  //队列满
        return ERROR;
    queue->data[queue->rear] = elem;  //队尾插入元素
    queue->rear = (queue->rear + 1) % maxn;  //队尾延伸，如果到了尽头，就转到数组最前面
    return RIGHT;
}

出队

Status Queue_Delete(struct Queue* queue, int* elem)  //出队，elem储存出队元素
{
    if (queue->front == queue->rear)  //队列空
        return ERROR;
    *elem = queue->data[queue->front];  //elem储存出队的元素
    queue->front = (queue->front + 1) % maxn;  //队头后移，如果到了尽头，就转到数组最前面
    return RIGHT;
}

队列长度

Status Queue_Length(struct Queue queue)  //队列长度
{
    return (queue.rear - queue.front + maxn) % maxn;  //公式
}

队列的链式存储

队头指针指向头结点，队尾指针指向终端结点

链式队列的结构及建立

struct QueueNode  //链式队列的结点
{
    int data;  //数值域
    struct QueueNode* next;  //指针域
};

struct LinkQueue  //链式队列
{
    struct QueueNode* front;  //队头指针
    struct QueueNode* rear;  //队尾指针
};

int main(void)
{
    struct LinkQueue Queue_link;  //建立链式队列
    struct QueueNode* head;  //定义头节点
    head->next = NULL;  //头结点指向空
    Queue_link.front = head;  //队头指针指向头结点
    Queue_link.rear = head;  //队尾指针指向头结点
}

入队

Status LinkQueue_Enter(struct LinkQueue* queue, int elem)  //入队，elem为入队元素
{
    struct QueueNode* q;  //q作为新的终端结点
    q = (struct QueueNode*)malloc(sizeof(struct QueueNode));  //为q分配内存空间
    if (!q)  //分配失败
        exit(OVERFLOW);  //在math.h头文件中，其值为 3，用以终止程序并上报系统缓存上溢
    q->data = elem;  //终端结点赋值
    q->next = NULL;  //终端结点指向空
    queue->rear->next = q;  //原队尾指针指向终端结点
    queue->rear = q;  //将终端节点赋给队尾指针
    return RIGHT;
}

出队

Status LinkQueue_Delete(struct LinkQueue* queue, int* elem)  //出队，elem储存出队元素
{
    struct QueueNode* p;  //p临时储存出队元素
    if (queue->front == queue->rear)  //空队列
        return ERROR;
    p = queue->front->next;  //出队的是头结点的下一个结点
    *elem = p->data;  //储存
    queue->front->next = p->next;  //头结点指向原来队首元素所指向 
    if (queue->rear == p)  //如果出队的是最后一个元素，在出队后队尾指针指向队头指针
        queue->rear = queue->front;
    free(p);  //释放内存
    return RIGHT;
}

栈

何谓栈者，就是数据存储进去，先入的反而后出的一种线性表（LIFO）

允许插入和删除的一端称为栈顶（top），另一端称之为栈底（bottom），不含任何数据元素的栈称为空栈

栈的顺序存储结构

结构定义及建栈

#define maxn 1000
#define ERROR 0
#define RIGHT 1
#define Status int

struct Stack  //栈的顺序存储结构
{
    int data[maxn];  //栈中储存元素
    int top;  //栈顶
};

int main(void)
{
    struct Stack Sqstack;  //建栈
    Sqstack.top = -1;  //一开始为空栈，栈顶位置为 -1
}

压栈

Status Stack_Push(struct Stack* stack, int elem)  //压栈
{
    if (stack->top == maxn - 1)  //栈满
        return ERROR;
    stack->data[++stack->top] = elem;  //栈顶上移，然后将elem赋给栈顶
    return RIGHT;
}

弹栈

Status Stack_Pop(struct Stack* stack, int* elem)  //弹出
{
    if (stack->top == -1)  //空栈
        return ERROR;
    *elem = stack->data[stack->top--];  //栈顶元素赋给elem，然后栈顶下降
    return RIGHT;
}

双栈共享空间

事实上，这种数据结构（双栈）的使用，一般是两个栈的空间需求相反时，是一个小技巧嗷

结构

struct DoubleStack  //双栈共享空间结构
{
    int data[maxn];  //栈中储存元素
    int top1;  //栈1栈顶
    int top2;  //栈2栈顶
};

压栈

Status DoubleStack_Push(struct DoubleStack* double_stack, int elem, int stack_number)  //压栈，elem为压入元素，stack_number为栈号参数，用以判断压入的是哪一个栈
{
    if (double_stack->top1 + 1 == double_stack->top2)  //栈满
        return ERROR;
    if (stack_number == 1)  //压入一号栈
        double_stack->data[++double_stack->top1] = elem;
    else if (stack_number == 2)  //压入二号栈
        double_stack->data[--double_stack->top2] = elem;
    return RIGHT;
}

弹栈

Status DoubleStack_Pop(struct DoubleStack* double_stack, int* elem, int stack_number)  //弹栈。elem储存弹出元素，stack_number为栈号参数，用以判断弹出的是哪一个栈
{
    if (stack_number == 1)  //弹出一号栈
    {
        if (double_stack->top1 == -1)  //空栈
            return ERROR;
        *elem = double_stack->data[double_stack->top1--];
    }
    else if (stack_number == 2)  //弹出二号栈
    {
        if (double_stack->top2 == maxn)  //空栈
            return ERROR;
        *elem = double_stack->data[double_stack->top2++];
    }
    return RIGHT;
}

栈的链式存储结构

链栈不需要头节点嗷，栈顶就是链头，当链栈为空时，其实就是top = NULL的时候

链栈的结构及建立

struct StackNode  //链栈结点
{
    int data;  //数值域
    struct StackNode* next;  //指针域
};

struct LinkStack  //链栈
{
    struct StackNode* top;  //链头
    int count;  //链中结点数
};

int main(void)
{
    struct LinkStack Stack_link;  //建栈
    Stack_link.top = NULL;  //初始化空栈
    Stack_link.count = 0;  //没有元素
}

进栈

Status LinkStack_Push(struct LinkStack* stack, int elem)  //进栈，elem为进栈元素
{
    StackNode* s = (struct StackNode*)malloc(sizeof(struct StackNode));  //为新结点（新栈顶）分配动态内存空间
    s->data = elem;  //赋值
    s->next = stack->top;  //新栈顶（新结点）指向原栈顶
    stack->top = s;  //新结点直接赋给栈顶指针
    stack->count++;  //栈中元素 + 1
    return RIGHT;
}

出栈

Status LinkStack_Pop(struct LinkStack* stack, int* elem)  //出栈，elem储存出栈元素
{
    StackNode* p;  //临时存放出栈元素，回收处理，防止内存泄漏
    if (stack->top == NULL)  //空栈
        return ERROR;
    *elem = stack->top->data;  //栈顶元素赋给elem
    p = stack->top;  //p指向栈顶
    stack->top = stack->top->next;  //栈顶指针指向栈顶元素指向
    free(p);  //释放内存
    stack->count--;  //栈中元素 - 1
    return RIGHT;
}

建立与输出

要实现其他基本操作，首先先要创建一个链表，并能够将其输出，此处将建立与输出整合到一起

尾插法创建单链表

这样生成的链表为正序

代码

#include <stdio.h>
#include <stdlib.h>

struct S
{
    int data;  //数值域
    struct S* next;  //指针域
};

int main(void)
{
    //这里是建立部分
    struct S* head;  //指向头结点的指针
    struct S* p, * q; //声明指向当前结点的指针（用以创建节点），指向上一结点的指针（用以链接）
    int n;  //定义一个整型的n作为这个链表的长度
    scanf_s("%d", &n);
    head = (struct S*)malloc(sizeof(struct S));  //为头结点申请分配动态内存空间
    head->next = NULL;  //将头结点中的指向下一节点的指针暂时性置为NULL
    q = head;  //将头结点的指针赋值给“上一结点”，对于第二个结点来说，它的上一节点就是头结点，这就是这样赋值的意义
    for (int i = 1; i <= n; i++)  //循环 n 次
    {
        p = (struct S*)malloc(sizeof(struct S));  //为当前结点申请分配动态内存空间（开辟一块新的“当前结点”）
        scanf_s("%d", &p->data);  //为数据域赋值
        p->next = NULL;  //将当前结点中的当中的指向下一结点的指针暂时性置为NULL，尽量这么做，因为这样最后一个结点的指针就能舒服地指向NULL
        q->next = p;  //将上一结点的指向下一结点的指针指向当前结点
        q = p;  //上一结点的指针指向当前结点为下一结点服务
    }
    //下面是输出部分
    struct S* visit;  //创建一个指向当前结点的指针
    visit = head->next;  //当前结点指向头结点指向的下一结点
    while (visit != NULL)  //如果当前结点的地址不为NULL的话
    {
        if (visit->next == NULL)  //判断当前结点屁股后面还有没有结点
            printf("%d\n", visit->data);
        else
            printf("%d ", visit->data);
        visit = visit->next;  //指向当前结点的指针指向下一结点
    }
    return 0;
}
//这里是头插法

查找

众所周知，链表没有像指针那样的下标，不支持随机访问，所以要查找某接待你，必须通过遍历的方式

就比如我希望定义一个函数，它可以实现：我输入一个值，它可以返还这个值在链表中的结点的位置，如果找不到就返还 -1

代码

int Search(struct S* head, int r)  //head 即是头结点，r 是要查找的值
{
    struct S* temp;  //定义用以遍历的指针
    temp = head;  //指针首先指向头结点
    int t = 1; //结点所在的位置
    while(temp->next != NULL)  //判断
    {
        temp = temp->next;  //指针后移
        if (r == temp->data)
            return t;  //其实这里返还的只有首位置，但是没大问题
        t++;
    }
    return -1;  //显然没有找到，返还 -1
}

修改

如果想要修改某一结点的数值域，首先要遍历找到这个结点，然后直接修改

代码

void Replace(struct S* head, int u, int v)  //head 即是头结点，u 是节点位置，v 是修改后的值
{
    struct S* temp;  //定义用以遍历的指针
    temp = head->next;  //指针首先指向开始的结点（头节点里无东西）
    for (int i = 1; i < u; i++)
        temp = temp->next;  //指针后移
    temp->data = v;  //直接修改
    return;
}

插入

有三种插入的方法：

  1. 头插
  2. 尾插
  3. 插入中间

但是插入的方法差不多，都是先将新结点的 next 指向插入节点后的那个结点，然后将插入结点前的那个结点的 next 指向插入结点

代码

void Insert(struct S* head, int u, int v)  //head 即是头结点，u 是插入的位置，v 是传入的数值域
{
    struct S* temp;  //定义用以遍历的指针
    temp = head;  //指针首先指向头结点
    int i = 0;
    while ((temp != NULL) && (i < u - 1))  //首先找到插入结点的上一个结点，即第(u - 1)个结点
    {
        temp = temp->next;
        i++;
    }
    if ((temp == NULL) || (i > u - 1))  //错误处理：链表为空或者不存在这个插入的位置
    {
        printf("Error!\n");  //输出错误信息
        return;
    }
    struct S* _new = (struct S*)malloc(sizeof(struct S));  //创建新结点_new
    _new->data = v;  //赋值给插入的新结点的数据域
    _new->next = temp->next;  //新结点的 next 指针指向插入位置后的结点
    temp->next = _new;  //插入位置前的结点的 next 指针指向新结点
    return;
}

删除

当删除结点的时候，主要是两步操作：（一定要按顺序！）

  1. 被删除结点的前一个结点的 next 指针指向被删除结点之后的结点（即将被删除结点“孤立出来”）
  2. 释放被删除结点的内存

代码

void Delete(struct S* head, int u, int* v)  //head 即是头结点，u 是删除的位置，v 用以保存被删除节点的数值域
{
    struct S* temp;  //定义用以遍历的指针
    temp = head;  //指针首先指向头结点
    int i = 0;
    while ((temp != NULL) && (i < u - 1))  //就像插入一样，首先找到插入结点的上一个结点，即第(u - 1)个结点
    {
        temp = temp->next;
        i++;
    }
    if ((temp == NULL) || (i > u - 1))  //错误处理：链表为空或者不存在这个插入的位置
    {
        printf("Error!\n");  //输出错误信息（就像插入一样，依然需要错误处理）
        return;
    }
    struct S* _del = temp->next;  //定义 _del 指向被删除的结点
    *v = _del->data;  //保存被删除结点的数值域
    temp->next = _del->next;  //被删除结点的上一个结点指向被删除结点的下一个节点
    free(_del);  //手动释放这个节点，防止内存泄漏
    _del = NULL;  //指向空
    return;
}

倒置

倒置算是这几种基本操作里面比较麻烦、难得理解的一个操作了

首先让头节点与第一个元素节点断开，但是要注意在断开之前需要用p指针指向第一个元素节点来保存第一个元素节点的位置，然后再断开。在这里有一个指针q指向一个指针域为空的节点，这个节点用来做为链表反转后的最后一个节点

让第二个元素节点的指针从指向第三个元素节点变为指向第一个元素节点，以此类推，直至指针p指向原链表最后一个元素

p指针指向NULL时，让原头节点的指针域指向原来最后一个元素节点。此时链表倒置已完成。

void Reserve(struct S* head)  //head 即是头结点
{
    struct S* temp;  //temp 暂时储存 p 所指向结点的下一个结点
    struct S* p, * q;  //p 用以遍历除开头结点之外的所有结点，并对它们进行操作
    p = head->next;  //p 指向 head 的下一个结点
    q = NULL;  //q 用以作为未操作完的新链表的临时头结点（除开头结点外的其他结点再进行操作时的头结点的）
    head->next = NULL;  //
    while (p != NULL)  //在 p 完成对尾结点的操作之后结束
    {
        temp = p->next;  //用temp 暂时储存 p 所指向结点的下一个结点，防止丢失后面的结点链
        p->next = q;  //p 指向 q
        q = p;  //q 指向 p 所指向的结点，即新的头结点
        p = temp;  //更新 p，使 p 指向下一个结点
    }
    head->next = q;  //头结点 head 指向 q
    return;
}

粉碎

粉碎也不能就只把头指针断掉，还要一个一个释放掉内存并清空嗷

void Crush(struct S* head)  //head 即使头结点
{
    struct S* temp;  //temp 暂时储存头结点所指的结点
    while (head != NULL)  //head 为空的时候结束
    {
        temp = head->next;
        free(head);  //释放
        head = temp;  //新的头结点（即前一个头结点的下一个结点）
    }
    return;
}

结构体

写这个主要是之前第一次用结构体AC了一道题，就巨兴奋，感觉想要记录下来，然后结构体在刷题中的确蛮好玩的，挺香，接下来是我之前刷过的两道题（应该会补充）

HDU1004

题目大意

有很多组测试，每一组首先给出一个数值，代表接下来出现的颜色个数，如果给出的数值为 0，则结束测试，每一组中，要你输出出现最多的颜色

理解

每种不一样的颜色使用一个结构体储存（储存名字和数量），每次输入都对储存不同颜色的结构体数组进行一次遍历，使用strcmp判断是否这个颜色已经“记录在案”，如果没有存在则使用strcpy将其储存下来（记得第一次特判（I think））

代码

#define _CRT_SECURE_NO_WARNINGS  //这个预处理命令加不加没必要，可以忽略一些警告的，比如VS2019中的安全性问题
#include <stdio.h>
#include <string.h>

char a[20];  //a 作为当前颜色而存在
char b[20];  //b 储存数量最多的颜色

struct s
{
    char u[20];  //储存颜色名字
    int v;       //储存当前该颜色的数量
};

int main(void)
{
    int n;  //多少组数据
    int t;  
    int m;  
    struct s k[1005];  
    while (scanf("%d", &n) != EOF && n != 0)
    {
        t = 0;  //记录颜色种类
        m = 0;  //每种颜色数量的最大值
        for (int i = 0; i < n; i++)  //这组共输入 n 个颜色
        {
            scanf_s("%s", &a, sizeof(a));  //更为安全的定义字符数组的方式
            if (t == 0)  //此处判断第一个输入的颜色
            {
                strcpy(k[0].u, a);  //把第一个输入的字符录入结构体数组中
                k[0].v = 1;  //该颜色数量初始化为 1
                t++;  //颜色种类 + 1
            }
            else
            {
                for (int j = 0; j < t; j++)  //遍历已有的颜色
                    if (strcmp(k[j].u, a) == 0)  //如果这种颜色已经出现过，就只需要增加这种颜色的数量
                    {
                        k[j].v++;
                        break;
                    }
                    else if (j == t - 1)
                    {
                        strcpy(k[t].u, a);  //如果这种颜色没有出现过，就增加一个颜色种类
                        k[t].v = 1;  //该颜色数量初始化为 1
                        t++;  //该颜色种类 + 1
                        break;
                    }
            }
        }
        for (int i = 0; i < t; i++)  //遍历所有颜色
            if (k[i].v > m)  //判断最大值
            {
                m = k[i].v;  //把当前颜色的最大值储存入 m
                strcpy(b, k[i].u);  //把当前数量最大的颜色储存入 b
            }
        printf("%s\n", b);  //输出最大值
    }
}

HDU2550

题目大意

一开始给出一个值，代表接下来的测试组数，每组先输入一个数，代表每组有多少对数据（A,B），每对数据的意思是有箭身长度为 A 的箭 B 枝，（这一道题是新生赛前的热身赛里的一道题，很离谱，热身赛全部题目的题干都巨尼玛中二，一道最简单的求距离的题可以说得玄乎其玄），然后要你按照箭身长度输出每种弓箭，每种弓箭之间还要空行（需要结构体排序）

就比如A = 3，B = 4的话就要输出这种箭 >+-+>

理解

这一道题的难点其实就只有结构体排序部分，只要结构体排序整出来了就很简单嗷

#include

struct S  //定义结构体

{

                    cout << "-";
                cout << "+>" << endl;
            }
            cout << endl;
        }
    }
    return 0;
}

九余定理

九余定理是一个和数学有关的知识，可以了解一下

首先要介绍九余数的概念：一个数的各位数字之和相加后得到的<10的数字称为这个数（如果相加结果大于9，则继续各位相加）

然后是九余定理的定义：一个数 x，他的每个位上的数加起来为 y，九余数为（y - 1）再对 9 求余然后加1，用 C 语言表示就是 ( y - 1 ) % 9 + 1

emmm，就是说一个数，如果我要求它的九余数，怎么求呢，这样求？

代码一

#include <stdio.h>

int main(void)
{
    int s, t;
    int n;
    while ((scanf("%d", &n) != EOF) && (n != 0))
    {
        s = 0;
        for (;;)
        {
            for (;;)
            {
                s = s + n % 10;
                n = n / 10;
                if (n == 0)
                    break;
            }
            if (s < 10)
                break;
            n = s;
            s = 0;
        }
        printf("%d\n", s);
    }
    return 0;
}

这是我第一次做的时候的代码，发现在处理较大的数时行不通，所以就换一种，再看看这种？
