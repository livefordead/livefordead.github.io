数据结构

队列

队列和栈刚好相反嗷，它是先进先出的线性表（FIFO）

允许删除的一端称为队头（front），允许插入的一端称为对尾（rear）

队列的顺序存储结构

由于数组存储空间的问题（即假溢出），我们使用的队列应该是头尾相接的循环队列，不然很容易就会超出数组范围

顺带一提队列长度计算公式：( rear - front + QueueSize ) % QueueSize

结构定义及初始化

#define maxn 1000
#define ERROR 0
#define RIGHT 1
#define Status int

struct Queue  //循环队列的顺序结构
{
    int data[maxn];  //队列中元素
    int front;
    int rear;
};

Status Queue_Init(struct Queue* queue)  //队列初始化
{
    queue->front = 0;  //队头归零
    queue->rear = 0;  //队尾归零
    return RIGHT;
}

int main(void)
{
    struct Queue SqQueue;
    if (Queue_Init(&SqQueue) != RIGHT)  //队列创建
        return;
}

入队

Status Queue_Enter(struct Queue* queue, int elem)  //入队，elem为入队元素
{
    if ((queue->rear + 1) % maxn == queue->front)  //队列满
        return ERROR;
    queue->data[queue->rear] = elem;  //队尾插入元素
    queue->rear = (queue->rear + 1) % maxn;  //队尾延伸，如果到了尽头，就转到数组最前面
    return RIGHT;
}

出队

Status Queue_Delete(struct Queue* queue, int* elem)  //出队，elem储存出队元素
{
    if (queue->front == queue->rear)  //队列空
        return ERROR;
    *elem = queue->data[queue->front];  //elem储存出队的元素
    queue->front = (queue->front + 1) % maxn;  //队头后移，如果到了尽头，就转到数组最前面
    return RIGHT;
}

队列长度

Status Queue_Length(struct Queue queue)  //队列长度
{
    return (queue.rear - queue.front + maxn) % maxn;  //公式
}

队列的链式存储

队头指针指向头结点，队尾指针指向终端结点

链式队列的结构及建立

struct QueueNode  //链式队列的结点
{
    int data;  //数值域
    struct QueueNode* next;  //指针域
};

struct LinkQueue  //链式队列
{
    struct QueueNode* front;  //队头指针
    struct QueueNode* rear;  //队尾指针
};

int main(void)
{
    struct LinkQueue Queue_link;  //建立链式队列
    struct QueueNode* head;  //定义头节点
    head->next = NULL;  //头结点指向空
    Queue_link.front = head;  //队头指针指向头结点
    Queue_link.rear = head;  //队尾指针指向头结点
}

入队

Status LinkQueue_Enter(struct LinkQueue* queue, int elem)  //入队，elem为入队元素
{
    struct QueueNode* q;  //q作为新的终端结点
    q = (struct QueueNode*)malloc(sizeof(struct QueueNode));  //为q分配内存空间
    if (!q)  //分配失败
        exit(OVERFLOW);  //在math.h头文件中，其值为 3，用以终止程序并上报系统缓存上溢
    q->data = elem;  //终端结点赋值
    q->next = NULL;  //终端结点指向空
    queue->rear->next = q;  //原队尾指针指向终端结点
    queue->rear = q;  //将终端节点赋给队尾指针
    return RIGHT;
}

出队

Status LinkQueue_Delete(struct LinkQueue* queue, int* elem)  //出队，elem储存出队元素
{
    struct QueueNode* p;  //p临时储存出队元素
    if (queue->front == queue->rear)  //空队列
        return ERROR;
    p = queue->front->next;  //出队的是头结点的下一个结点
    *elem = p->data;  //储存
    queue->front->next = p->next;  //头结点指向原来队首元素所指向 
    if (queue->rear == p)  //如果出队的是最后一个元素，在出队后队尾指针指向队头指针
        queue->rear = queue->front;
    free(p);  //释放内存
    return RIGHT;
}
