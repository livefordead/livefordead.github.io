注册

注册，首先我们需要在用户注册成功的同时创建一个专属于该用户的文本文档用以储存该用户的所有信息（下面就不采用调用函数的形式了）

代码

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

struct User
{
    char name[20] = "";  //用户名
    char word[20] = "";  //密码
};

FILE* fp;  //这个你学文件里面应该有滴

struct User user;

int main(void)
{
    scanf_s("%s", user.name, sizeof(user.name));  //安全性
    fp = fopen(user.name, "r");  //以只读的方式打开这个文件，fp指向以user.name为名的文件，如果没有这个文件，则fp指向NULL
    if (fp != NULL)  //说明已经存在这个文件，就不可以用这个用户名注册，进行你想要的操作
    {

    }
    else  //说明这个用户名可用，其实在前面还可以加上判断用户名长度
    {
        //然后是输入密码并确认密码的操作
    }
    fp = fopen(user.name, "w");  //user.name写入文件，user.name是文件名，要是没有成功，fp指向NULL
    if (fp == NULL)  //显然注册失败
    {

    }
    else  //注册成功
    {
    //任意发挥，比如输出一些话
        if (fwrite(&user, sizeof(User), 1, fp) != 1)  /*第一个是指向结构体的地址，第二个是写入内容的大小，
                                                   第三个是数据项的个数（这里我只要写入一个结构体，所以是 1），第四个是目标文件指针
                                                   返还值是实际的写入数据项的个数
                                                 */
        {
        //要是返还值不是 1，就说明写入文件错误，可以输出一个提示，然后退出
        }
    }
//到此为止，这就是注册的基本架构啦
}

登录

要登陆，首先是要建立在这个用户创建了的基础上，就是说我要可以找到属于这个用户的文本文档，要是找不到，就说明不存在这个用户，就可以提示并结束，然后找到了用户之后需要比较原始密码与输入密码（可以加入输错三次锁死或退出之类的）

代码

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>

struct User
{
    char name[20] = "";  //用户名
    char word[20] = "";  //密码
};

FILE* fp;  //这个你学文件里面应该有滴

struct User user;

int main(void)  // 好啦，接下来是在有了注册的基础上
{
    char _id[20]; //用以输入的用户名
    char _pass[20];  //用以输入的密码
    scanf_s("%s", _id, sizeof(_id));
    fp = fopen(_id, "r+");  //只读，看是不是存在这个用户名
    if (fp == NULL)  //显然不存在
    {
        //进行你想要的操作
    }
    fread(&user, sizeof(User), 1, fp);  //从fp所指向的文件处读取user的信息
    scanf_s("%s", &_pass, sizeof(_pass));
    if (strcmp(user.word, _pass) != 0)  //比较正确密码与输入的密码
    {
        //输错了，咋办
    }
    else
    {
        //对啦，登陆成功啦
    }
}

HDU2054 !

A==B?

吐槽

HDU2054这道题，好久之前做的一道挺有意思的水题，这一题从暑假开始一直A到开学后十几天，终于是做出来了，一开始太天真地以为它真的就是道下饭的题（虽然的确挺水），结果自己的操作是真的下饭，很离谱，但是后来还是莫名想通了，去搜了自己要的东西就顺利做了出来。

题解

虽然是比较两个数是否相等，但是并不是让你比较两个整形，这本质上其实还是道大数题，CSDN、博客园里面有蛮多人的代码都考虑太多，主要有这几种情况：

· 要考虑小数点和其后的无效零

· 不要考虑前导零

· 不要考虑正负号

所以可以开一个字符数组，分别检索小数点存在与否（可以使用 strstr 或者 for 循环遍历查找），然后将其后的 0 全部转化为 ‘\0’ （终止符），最后将小数点转化为 ‘\0’ （终止符），最后用 strcmp比较即可

代码

#include <stdio.h>
#include <string.h>

char a[100000];
char b[100000];

void S(char s[])
{
    int l;
    l = strlen(s);  //取字符数组的长度
    if(strstr(s, "."))  //使用strstr查找是否存在小数点
    {
        for(int i = l - 1; s[i] == '0'; i--)  //从最后一位向前去除所有后置无效零
        {
            s[i] = '\0';
            l--;
        }
    }
    if(s[l - 1] == '.')  //去除小数点
        s[l - 1] = '\0';
}

int main(void)
{
    while(scanf_s("%s %s", &a, &b, sizeof(a), sizeof(b)) != EOF)
    {
        S(a);
        S(b);
        if(strcmp(a, b) == 0)  //比较
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}

Rule Taking

Posted on 2019-12-16

尺取法 (Rule Taking)

常见用法

一个数组中寻找长度最（小 / 大）的一段满足和最大或者大于某一个定值

主要思想

当 a1 , a2 , a3 ...... an 满足和 >= S，得到一个区间长度 n，那么去掉开头 a1，剩下 a2 , a3 ... 判断是否满足 >= S，如果满足，那么区间长度更新，如果不满足，那么尾部向后延展，判断 a2 , a3 ...... a (n + 1) 是否满足条件，重复这样的操作

理解与思考

当一个区间满足条件时，那么去掉区间开头第一个数，得到新区间，判断新区间是否满足条件，如果不满足条件，那么区间末尾向后扩展，直到满足条件为止，这样就得到了很多满足条件的区间，再根据题意，在此之间进行选择，比如区间最长，区间最短啥的，这样跑一遍下来，时间复杂度只有 O (n)

比如

给出 T 组数据，每组数据有 n 个数，并且每次给你一个 s 值，求最小的满足相加大于 s 的连续子序列的长度

详情请前往POJ3061

下面贴代码（C语言）

代码

#include<stdio.h>

int a[100005]; //定义一个数组来储存每组的数

int main(void)
{
    int T; 
    scanf("%d", &T);
    while (T--)
    {
        int n, s;
        scanf("%d %d", &n, &s);
        for (int i = 1; i <= n; i++)
            scanf("%d", &a[i]);  //赋值
        int k = n + 1;
        int u = 0;
        int v = 0;
        int m = 0; //清零操作
        for (;;)
        {
            while (m < s && v < n)
                m = m + a[++v];  //从第一个数开始增加，直到和大于等于 s 或者在 v 大于等于 n 的时候还是要小，就结束循环
            if (m < s)  //判断和是否大于s,其实就是看整个数段总和是否大于s，如果不就跳出循环
                break;
            if (k > v - u)  // u 作为数段的前标，v 作为数段的后标，v - u 就是数段的长度
                k = v - u; //取小的
            m = m - a[u++]; //在确定 s 比现在的和小后，从前面开始缩短数段，再进行循环
        }
        if (k == n + 1) //这是所有数的和都比 s 小的情况
            k = 0;
        printf("%d\n", k);  //输出
    }
    return 0;
}
